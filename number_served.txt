# Create a class called Restaurant
class Restaurant:
    # Initialize the restaurant attributes
    def __init__(self, restaurant_name, cuisine_type):
        self.restaurant_name = restaurant_name
        self.cuisine_type = cuisine_type

        # Attribute to track number of customers served
        self.number_served = 0

    # Method that prints the restaurant details
    def describe_restaurant(self):
        print("Restaurant Information:")
        print(f"Name: {self.restaurant_name}")
        print(f"Cuisine Type: {self.cuisine_type}")

    # Method that indicates the restaurant is open
    def open_restaurant(self):
        print("The restaurant is now open.")

    # Method to set the number of customers served
    def set_number_served(self, number):
        self.number_served = number

    # Method to increment the number of customers served
    def increment_number_served(self, number):
        self.number_served += number


# Create an instance of the Restaurant class
restaurant = Restaurant("La Bella Italia", "Italian")

# Print the initial number of customers served
print(f"Customers served: {restaurant.number_served}")

# Change the value and print again
restaurant.number_served = 12
print(f"Customers served: {restaurant.number_served}")

# Use set_number_served() and print again
restaurant.set_number_served(30)
print(f"Customers served: {restaurant.number_served}")

# Increment the number of customers served
restaurant.increment_number_served(18)
print(f"Customers served after increment: {restaurant.number_served}")
