# Create a class called User
class User:
    # Initialize the user attributes
    def __init__(self, first_name, last_name, age, email, city):
        self.first_name = first_name
        self.last_name = last_name
        self.age = age
        self.email = email
        self.city = city

        # Attribute to track login attempts
        self.login_attempts = 0

    # Method that prints a summary of the user's information
    def describe_user(self):
        print("User Information:")
        print(f"Name: {self.first_name} {self.last_name}")
        print(f"Age: {self.age}")
        print(f"Email: {self.email}")
        print(f"City: {self.city}")

    # Method that prints a personalized greeting
    def greet_user(self):
        print(f"Hello {self.first_name}, welcome back!\n")

    # Method that increments login attempts by 1
    def increment_login_attempts(self):
        self.login_attempts += 1

    # Method that resets login attempts to 0
    def reset_login_attempts(self):
        self.login_attempts = 0


# Create an instance of the User class
user = User(
    "Aaron Ludwig",
    "Altar",
    21,
    "2320616@ub.edu.ph",
    "Malvar"
)

# Call increment_login_attempts() several times
user.increment_login_attempts()
user.increment_login_attempts()
user.increment_login_attempts()

# Print the current number of login attempts
print(f"Login Attempts: {user.login_attempts}")

# Reset the login attempts
user.reset_login_attempts()

# Print the login attempts again to confirm reset
print(f"Login Attempts after reset: {user.login_attempts}")
